using namespace std;

int main() {
	string arr;
	char* pat = new char[1000];
	int idx = 0;
	getline(cin, arr);

	for (int i = 0; i < arr.length(); i++) {
		if (arr[i] == 'U' || arr[i] == 'C' || arr[i] == 'P') {
			pat[idx] = arr[i];
			idx++;
		}
	}

	int j = 0, count = 0;
	char ucpc[5] = { 'U', 'C', 'P', 'C' };

	for (int i = 0; i < strlen(pat); i++) {
		for (j = 0; j < 4; j++) {
			if (pat[i] == ucpc[j]) {
				count += 1;
				i++;
				if (count == 4) {
					cout << "I love UCPC\n";
					return 0;
				}
			}
			else {
				i++;
				break;
			}
		}
	}	
	cout << "I hate UCPC\n";
	return 0;
}
