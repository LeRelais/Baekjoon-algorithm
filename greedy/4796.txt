using namespace std;

int main() {
	int p, l, v;
	int count, days = 0, idx = 1;
	while (true) {
		cin >> l >> p >> v;
		if (p == 0)
			break;
		count = v / p;
		v %= p;
		days = count * l + v;
		printf("Case %d: %d\n", idx, days);
		idx++;
	}
}
